from fastapi import FastAPI
from google_genai import Client

app = FastAPI()

class Config:
    GEMINI_API_KEY = os.getenv("GEMINI_API_KEY", "")
    GEMINI_MODEL   = os.getenv("GEMINI_MODEL", "models/gemini-2.0-flash-001")

_gemini_client = Client(api_key=Config.GEMINI_API_KEY)

def _gemini_score_sync(prompt: str) -> str:
    print("[GEMINI MODEL]", Config.GEMINI_MODEL)
    resp = _gemini_client.models.generate_content(
        model=Config.GEMINI_MODEL,
        prompt=prompt,
        temperature=0.7,
        max_tokens=256
    )
    return resp.text
